import ConsistencyGO.Utils.Compact

def Constant {α β : Type*} (f : α → β) := ∃ c, ∀ x, f x = c

def ConstantOn {α β : Type*} (A : Set α) (f : α → β) := ∃ c, ∀ x ∈ A, f x = c
